## Architecture Reference

### AttentionGRU (adapted from PrunedAttentionGRU.py)
Replace CustomGRU with nn.GRU (cuDNN optimized). Replace MaskedAttention with standard nn.Linear attention.

```python
class AttentionGRU(nn.Module):
    def __init__(self, input_dim=52, hidden_dim=128, attention_dim=32, output_dim=4, num_classes=4):
        super().__init__()
        self.gru = nn.GRU(input_dim, hidden_dim, batch_first=True)
        # Additive attention: query+key tanh -> context vector -> softmax -> weighted sum
        self.attention = nn.Sequential(
            nn.Linear(hidden_dim, attention_dim),
            nn.Tanh(),
            nn.Linear(attention_dim, 1, bias=False)
        )
        self.fc = nn.Linear(hidden_dim, output_dim)
    
    def forward(self, x):
        # x: (batch, time_steps, 52)
        gru_out, _ = self.gru(x)  # (batch, time_steps, hidden_dim)
        attn_scores = self.attention(gru_out).squeeze(-1)  # (batch, time_steps)
        attn_weights = F.softmax(attn_scores, dim=1)  # (batch, time_steps)
        context = (gru_out * attn_weights.unsqueeze(-1)).sum(dim=1)  # (batch, hidden_dim)
        output = self.fc(context)  # (batch, output_dim)
        return output
```
